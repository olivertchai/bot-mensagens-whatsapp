FROM php:8.4-fpm
ENV TZ=’America/Sao_Paulo’

# Copia o executável do Composer diretamente da imagem oficial, 
# evitando a instalação manual via script.
COPY --from=composer:2 /usr/bin/composer /usr/local/bin/composer

RUN apt-get update && apt-get install -y --no-install-recommends\
    #libxml2-dev: Necessária para processar arquivos XML e APIs SOAP.
    libxml2-dev \  
    #libcurl4-openssl-dev: Permite que sua aplicação faça requisições HTTP (comunicações com outras APIs) de forma segura.
    libcurl4-openssl-dev\
    #libzip-dev: Permite que o PHP manipule arquivos .zip.
    libzip-dev \
    #libonig-dev: Biblioteca essencial para o processamento de expressões regulares e multibyte (texto complexo).
    libonig-dev \
    # É aqui que você dá "superpoderes" ao seu binário do PHP, instalando as extensões necessárias para o seu projeto.
    && docker-php-ext-install\
    pdo_mysql \
    xml \
    curl \
    zip \
    bcmath \
    mbstring \
    && apt-get clean && rm -rf /var/lib/apt/lists/*

#É o "GPS" do seu Dockerfile. Ele define o diretório de trabalho dentro do container, onde os comandos subsequentes serão executados.
WORKDIR /app

COPY src/ .

#Este comando baixa todas as bibliotecas PHP que seu projeto precisa (definidas no composer.json).
RUN composer install --no-dev --optimize-autoloader

EXPOSE 8000

# Este comando é o "ponto de partida" do seu container. Ele inicia o servidor de desenvolvimento do Laravel, permitindo que sua aplicação seja acessível na porta 8000.
CMD ["php", "artisan", "serve", "--host=0.0.0.0"]
